import discord
TOKEN = ' '
INTENTS = discord.Intents.all()
client = discord.Client(intents = INTENTS)

@client.event   #event는 실행되는 동안 발생했을때
async def on_ready():   #on_ready 봇이 off->on되엇을때
    await client.change_presence(status=discord.Status.online, activity=discord.Game("봇의 상태"))

# 봇 실행
client.run(TOKEN)
